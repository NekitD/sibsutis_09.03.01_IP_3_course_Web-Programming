
class Карточка :
Описание(_Описание)
{
    public:
        Карточка(string _Описание);
        operator<<(Карточка* this){
            cout << Описание << endl;
        }
    private:
        string Описание
}

class Эмоция : public Карточка :
Описание(_Описание),
Название(_Название)
{
    Эмоция(string _Название, string _Описание);
    operator<<(Карточка* this){
        cout << Название << endl;
        cout << Описание << endl;
    }
    private:
        string Название
}


class Игрок 
{
    Эмоция* Эмоция 
    vector<Карточка*> Навыки
    vector<Карточка*> Профессии
}
